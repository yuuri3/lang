\begin{tabular}{|l|c|c|}
    \hline
    \textbf{文法範疇} & \textbf{類別名詞} & \textbf{一般名詞} \\
    \hline
    \textbf{性} & \textbf{〇} & \textbf{×} \\
    \textbf{数} & \textbf{〇} & \textbf{×} \\
    \textbf{格} & \textbf{〇} & \textbf{〇} \\
    \hline
\end{tabular}
