\paragraph{/ai/}
二重母音/ai/は\textipa{[a\textsubarch{I}]}～\textipa{[aj]}と発音される。

\begin{tabular}{lll}
    baia & \textipa{[ba\textsubarch{I}a]} & マチェット \\
    dai & \textipa{[da\textsubarch{I}]} & 嗅ぐ \\
\end{tabular}

\paragraph{/au/}
二重母音/au/は\textipa{[A\textsubarch{U}]}～\textipa{[Aw]}と発音される。

\begin{tabular}{lll}
    au & \textipa{[A\textsubarch{U}]} & 高い \\
    nauba & \textipa{[nA\textsubarch{U}\super{m}bA]} & 弓 \\
\end{tabular}

\paragraph{/ei/}
二重母音/ei/は\textipa{[e\textsubarch{I}]}～\textipa{[ej]}と発音される。

\begin{tabular}{lll}
    atei & \textipa{[ate\textsubarch{I}]} & しゃがむ \\
\end{tabular}

\paragraph{/ou/}
二重母音/ou/は\textipa{[o\textsubarch{U}]}～\textipa{[ow]}と発音される。

\begin{tabular}{lll}
    nou & \textipa{[no\textsubarch{U}]} & 狩る \\
    pou & \textipa{[po\textsubarch{U}]} & 笑う \\
\end{tabular}
